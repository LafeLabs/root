svgdir = "../symbol/physics/circuits/svg/";
var httpc = new XMLHttpRequest();
httpc.onreadystatechange = function() {
    if (this.readyState == 4 && this.status == 200) {
        data = this.responseText;
        svginit();
    }
};

httpc.open("GET", "loadsvgfeed.php?svgdir=" + svgdir, true);
httpc.send();

function svginit(){
    var svgfilelist = data.split(",");
    for(var index = 0;index < svgfilelist.length;index++){
        if(svgfilelist[index].length > 3){
            var newimg = document.createElement("IMG");
            newimg.src = svgdir + svgfilelist[index];
            document.getElementById("scrolldiv").appendChild(newimg);
            newimg.onclick = function(){
                document.getElementById("urlinput").value = this.src;
            }
        }
    }
}


</script>
