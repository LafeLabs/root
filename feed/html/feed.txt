<script>

    jsondata = JSON.parse(document.getElementById("jsondata").innerHTML);
    
    for(var index = 0;index < jsondata.length;index++){
        var newdiv = document.createElement("DIV");
        newdiv.className = "unit";
        document.getElementById("feed").appendChild(newdiv);
        var newimg1 = document.createElement("IMG");
        newimg1.src = jsondata[index].bottomimage;
        newimg1.className = "bottomimage";
        var newimg2 = document.createElement("IMG");
        newimg2.src = jsondata[index].topimage;
        newimg2.className = "topimage";
        newdiv.appendChild(newimg1);
        newdiv.appendChild(newimg2);
        phi = jsondata[index].phi;
        alpha = Math.cos(phi)*Math.cos(phi);
        beta = Math.sin(phi)*Math.sin(phi);
        newimg2.style.opacity = alpha.toString();        
        newimg1.style.opacity = beta.toString();        

    }
    

</script>
