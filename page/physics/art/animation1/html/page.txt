<script>
function myMove() {
  var elem = document.getElementById("animate");   
  var pos = 0;
  var count = 0;
  var id = setInterval(frame, 1);
  function frame() {
    if (pos == 350) {
		pos = 0;
        count++;
	} else {
      pos +=2; 
      elem.style.top = pos + 'px'; 
      elem.style.left = pos + 'px'; 
    }
    if(count ==5){
		pos = 0;
      elem.style.top = pos + 'px'; 
      elem.style.left = pos + 'px';         
       clearInterval(id);
    }
  }
}
</script>
